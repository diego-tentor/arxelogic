def recursive_mass(m0, a, n_steps, sign=1):
    """
    Compute mass via recursive formula: m_{k+1} = a*m_k + sign*π
    
    Parameters:
    - m0: Initial mass
    - a: Amplification factor
    - n_steps: Number of iterations
    - sign: Sign of π term (+1 or -1)
    
    Returns:
    - Final mass and history
    """
    m = m0

        m = a * m + sign * np.pi
        history.append(m)
